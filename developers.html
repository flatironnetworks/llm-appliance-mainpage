Quick Start

Switch from cloud to local with a single line change

Before (Cloud)
from openai import OpenAI

client = OpenAI(
    api_key="sk-...",
    # Default: api.openai.com
)

response = client.chat.completions.create(
    model="gpt-4",
    messages=[{"role": "user", "content": "Hello!"}]
)
After (LLM Appliance)
from openai import OpenAI

client = OpenAI(
    api_key="your-api-key",
    base_url="http://llm-appliance.local/v1"  # Just add this line
)

response = client.chat.completions.create(
    model="llama-3.3-70b",
    messages=[{"role": "user", "content": "Hello!"}]
)
